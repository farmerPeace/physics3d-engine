#include <iostream>
#include <cmath>
#include <cassert>
#include <type_traits>
#include <stdexcept>
#include <limits>
#include <algorithm>

class Vector3D {
    private:
        float x_, y_, z_;
    
    public:
        // Getters
        float x() const { return x_; }
        float y() const { return y_; }
        float z() const { return z_; }

        // Setters
        void set_x(float x) { x_ = x; }
        void set_y(float y) { y_ = y; }
        void set_z(float z) { z_ = z; }

        // Inicializadores
        Vector3D(): x_(0), y_(0), z_(0) {}
        Vector3D(float x, float y, float z): x_(x), y_(y), z_(z) {}

        // Operaciones con vectores
        // Suma de dos vectores
        Vector3D &operator += (const Vector3D &rhs) {
            x_ += rhs.x_;
            y_ += rhs.y_;
            z_ += rhs.z_;

            return *this;
        }
        
        Vector3D operator-(const Vector3D& rhs) const {
            return Vector3D(x_ - rhs.x_, y_ - rhs.y_, z_ - rhs.z_);
        }
    
    

        // Resta de dos vectores
        Vector3D &operator -= (const Vector3D &rhs) {
            x_ -= rhs.x_;
            y_ -= rhs.y_;
            z_ -= rhs.z_;

            return *this;        
        }

        Vector3D operator+(const Vector3D& rhs) const {
            return Vector3D(x_ + rhs.x_, y_ + rhs.y_, z_ + rhs.z_);
        }

        // Negativo de un vector
        Vector3D operator -() const {
            return Vector3D(-x_, -y_, -z_);
        }

        // Operadores de comparación
        bool operator == (const Vector3D &rhs) const {
            return x_ == rhs.x_ && y_ == rhs.y_ && z_ == rhs.z_;
        }

        bool operator != (const Vector3D &rhs) const {
            return !(*this == rhs);
        }

        // Indices
        float operator[] (int index) const {
            if (index == 0) return x_;
            if (index == 1) return y_;
            if (index == 2) return z_;

            throw std::out_of_range("Indice Vector3D fuera de rango");
        }

        float &operator [] (int index) {
            if (index == 0) return x_;
            if (index == 1) return y_;
            if (index == 2) return z_;

            throw std::out_of_range("Indice Vector3D fuera de rango");
        }

        // Multiplicación escalar
        template <typename Scalar>
        Vector3D &operator *= (const Scalar &scalar) {
            static_assert(std::is_arithmetic_v<Scalar>, "Scalar debe ser tipo aritmetico");

            x_ *= static_cast<float>(scalar);
            y_ *= static_cast<float>(scalar);
            z_ *= static_cast<float>(scalar);

            return *this;
        }

        // División escalar
        template <typename Scalar>
        Vector3D &operator /= (const Scalar &scalar) {
            static_assert(std::is_arithmetic_v<Scalar>, "Scalar debe ser tipo aritmetico");
            
            const float scalar_f = static_cast<float>(scalar);

            if (std::abs(scalar_f) < std::numeric_limits<float>::epsilon()) {
                throw std::domain_error("División por casi cero en un Vector3D");
            }

            const float inv = 1.0f / scalar_f;
            x_ *= inv;
            y_ *= inv;
            z_ *= inv;

            return *this;
        }

        // Producto Punto
        float Dot (const Vector3D &rhs) const {
            return x_ * rhs.x_ + y_ * rhs.y_ + z_ * rhs.z_;
        }

        // Producto Cruz
        Vector3D Cross(const Vector3D &rhs) const {
            return Vector3D (
            y_ * rhs.z_ - z_ * rhs.y_,
            z_ * rhs.x_ - x_ * rhs.z_,
            x_ * rhs.y_ - y_ * rhs.x_
            );
        }

        // Magnitud y magnitud al cuadrado
        float Magnitude() const {
            return std::sqrt(x_ * x_ + y_ * y_ + z_ * z_);     
        }

        float SquareMagnitude() const {
            return x_ * x_ + y_ * y_ + z_ * z_;
        }

        // Normalización de un vector
        Vector3D &Normalize() {
            const float sq_mag = SquareMagnitude();

            if (sq_mag > std::numeric_limits<float>::epsilon()) {
                const float inv_mag = 1.0f / std::sqrt(sq_mag);
                
                x_ *= inv_mag;
                y_ *= inv_mag;
                z_ *= inv_mag;
                
            } else {
                x_ = 0.0f;
                y_ = 0.0f;
                z_ = 0.0f;
            }

            return *this;
        }

        Vector3D Normalized() const {
            Vector3D result = *this;
            result.Normalize();
            return result;
        }

        // Distancia entre dos vectores
        static float Distance(const Vector3D& a, const Vector3D& b) {
            return (b - a).Magnitude();
        }

        static float SquaredDistance(const Vector3D& a, const Vector3D& b) {
            return (b - a).SquareMagnitude();
        }
};


// Multiplicación escalar por ambos lados
template <typename Scalar>
inline Vector3D operator *(const Scalar &scalar, Vector3D vect) {
    vect *= scalar;
    return vect;
}

template <typename Scalar>
inline Vector3D operator *(Vector3D vect, const Scalar &scalar) {
    vect *= scalar;
    return vect;
}

// División escalar
template<typename Scalar>
inline Vector3D operator / (Vector3D vect, const Scalar& scalar) {
    vect /= scalar;
    return vect;
}
